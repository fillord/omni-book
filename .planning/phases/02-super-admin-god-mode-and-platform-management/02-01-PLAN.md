model Announcement {
  id        String   @id @default(cuid())
  title     String
  body      String
  isActive  Boolean  @default(true)
  createdAt DateTime @default(now())
  updatedAt DateTime @updatedAt

  @@index([isActive])

**3. Add Notification model:**
```prisma
model Notification {
  id        String   @id @default(cuid())
  tenantId  String
  message   String
  read      Boolean  @default(false)
  createdAt DateTime @default(now())
  tenant    Tenant   @relation(fields: [tenantId], references: [id], onDelete: Cascade)

  @@index([tenantId])
  @@index([tenantId, read])
}
```

**4. Add AuditLog model:**
```prisma
model AuditLog {
  id        String   @id @default(cuid())
  tenantId  String
  eventType String
  details   Json     @default("{}")
  createdAt DateTime @default(now())
  tenant    Tenant   @relation(fields: [tenantId], references: [id], onDelete: Cascade)

  @@index([tenantId])
  @@index([eventType])
  @@index([createdAt])
}
```
