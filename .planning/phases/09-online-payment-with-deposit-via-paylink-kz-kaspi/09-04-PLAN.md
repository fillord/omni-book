**F. Create `WaitingForPaymentScreen` component** inside the same file (before the main component export, or as a nested function component):

```typescript
function WaitingForPaymentScreen({ bookingId, onReset }: { bookingId: string; onReset: () => void }) {
  const { t } = useI18n()
  const [secondsLeft, setSecondsLeft] = useState(10 * 60) // 10 minutes
  const [expired, setExpired] = useState(false)

  useEffect(() => {
    if (secondsLeft <= 0) {
      setExpired(true)
      return
    }
    const timer = setInterval(() => {
      setSecondsLeft((prev) => {
        if (prev <= 1) {
          setExpired(true)
          return 0
        }
        return prev - 1
      })
    }, 1000)
    return () => clearInterval(timer)
  }, [secondsLeft])

  const minutes = Math.floor(secondsLeft / 60)
  const seconds = secondsLeft % 60
  const timeDisplay = `${String(minutes).padStart(2, '0')}:${String(seconds).padStart(2, '0')}`

  if (expired) {
    return (
      <div className="text-center space-y-4 p-6 rounded-2xl bg-[var(--neu-bg)] neu-raised">
        <div className="text-4xl">&#9200;</div>
        <h3 className="text-lg font-semibold text-foreground">
          {t('booking', 'paymentExpired') || 'Время ожидания оплаты истекло. Запись отменена.'}
        </h3>
        <p className="text-sm text-muted-foreground">
          {t('booking', 'paymentExpiredSub') || 'Запись автоматически отменена. Вы можете записаться снова.'}
        </p>
        <button
          onClick={onReset}
          className="mt-4 px-6 py-2 rounded-xl bg-[var(--neu-bg)] neu-raised text-sm font-medium hover:opacity-80 transition-opacity"
        >
          {t('booking', 'bookAgain') || 'Записаться снова'}
        </button>
      </div>
    )
  }

  return (
    <div className="text-center space-y-4 p-6 rounded-2xl bg-[var(--neu-bg)] neu-raised">
      <div className="text-4xl">&#128179;</div>
      <h3 className="text-lg font-semibold text-foreground">
        {t('booking', 'waitingForPayment') || 'Ожидание оплаты'}
      </h3>
      <p className="text-sm text-muted-foreground">
        {t('booking', 'waitingInstructions') || 'Мы выставили счет в ваше приложение Kaspi. Оплатите в течение 10 минут.'}
      </p>
      <div className="text-3xl font-mono font-bold text-foreground neu-inset inline-block px-6 py-3 rounded-xl bg-[var(--neu-bg)]">
        {timeDisplay}
      </div>
      <p className="text-xs text-muted-foreground">
        {t('booking', 'paymentCountdown') || 'Осталось времени:'}
      </p>
    </div>
  )
}
```
